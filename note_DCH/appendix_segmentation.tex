\begin{equation}
  \Delta a = L \cdot \tan(\epsilon)
\end{equation}


\begin{equation}
  \alpha = 2 \cdot \arcsin\left({{L \cdot \tan(\epsilon)} \over {2
        \cdot R}}\right)
\end{equation}


\begin{itemize}
\item $\phi_{start}=\phi$
\item $\phi_{end}=\phi+\alpha$
  \begin{itemize}
  \item If $\epsilon<0 \Rightarrow \alpha<0$
  \end{itemize}
\end{itemize}

\begin{equation}
  \vec{w}_{start} = \begin{pmatrix}
    R \cos(\phi_{start}) \\
    R \sin(\phi_{start}) \\
    L/2
  \end{pmatrix}
\end{equation}

\begin{equation}
  \vec{w}_{end} = \begin{pmatrix}
    R \cos(\phi_{end}) \\
    R \sin(\phi_{end}) \\
    -L/2
  \end{pmatrix}
\end{equation}
